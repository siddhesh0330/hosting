//2B : Increment Decrement Counter

pragma solidity >=0.8.2 <0.9.0;
contract Counter {
 uint256 public num;
 function input(uint256 number) public{
 num = number;
 }
 function increment() public view returns (uint256) {
 return num+1;
 }
 function decrement() public view returns (uint256) {
 return num-1;
 }
}
